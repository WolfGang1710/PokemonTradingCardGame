80void Player::attack(const size_t attackerIndex, const size_t attackIndex, Player &opponent, const size_t enemyPokemonIndex) {
81 /*
82 * @brief: attack an opponent's PokemonCard
83 * @details: if the attackerIndex, attackIndex and enemyPokemonIndex are valid, attack the opponent's PokemonCard at the given index with the PokemonCard at the given index. If the opponent's PokemonCard is eliminated, remove it from the opponent's actionCards vector
84 * @param: `attackerIndex`: index of the PokemonCard to attack with
85 * @param: `attackIndex`: index of the attack to use
86 * @param: `opponent`: reference to the opponent
87 * @param: `enemyPokemonIndex`: index of the PokemonCard to attack
88 * @return: none
89 *
90 */
91 if (attackerIndex < actionCards.size() && enemyPokemonIndex < opponent.actionCards.size()) {
92 PokemonCard* attacker = dynamic_cast<PokemonCard*>(actionCards[attackerIndex]);
93 PokemonCard* enemyPokemon = dynamic_cast<PokemonCard*>(opponent.actionCards[enemyPokemonIndex]);
94
95 // Vérifier si les cartes sont du bon type
96 if (attacker && enemyPokemon) {
97 // Vérifier si l'indice de l'attaque est valide
98 if (attackIndex < attacker->getNumberOfAttacks()) {
99 // Récupérer les informations sur l'attaque
100 auto& attack = attacker->getAttack(attackIndex);
101 int energyCost = get<0>(attack);
102 int currentEnergy = get<1>(attack);
103
104 // Vérifier si le Pokémon attaquant a assez d'énergie
105 if (currentEnergy >= energyCost) {
106 // Effectuer l'attaque
107 cout << endl << playerName << " attacking " << opponent.playerName << "'s Pokemon "
108 << opponent.actionCards[enemyPokemonIndex]->getCardName()
109 << " with the Pokemon " << actionCards[attackerIndex]->getCardName()
110 << " with its attack: " << get<2>(attack) ;
111
112 // Appliquer les dégâts au Pokémon attaqué
113 enemyPokemon->receiveDamage(get<3>(attack));
114 cout << endl << "Reducing " << get<3>(attack) << " from " << opponent.playerName << "'s Pokemon's HP" ;
115
116 // Vérifier si le Pokémon attaqué est éliminé
117 if (enemyPokemon->getHP() <= 0) {
118 cout << endl << opponent.actionCards[enemyPokemonIndex]->getCardName() << " fainted.";
119 opponent.actionCards.erase(opponent.actionCards.begin() + enemyPokemonIndex);
120 } else {

122 }
123 } else {
